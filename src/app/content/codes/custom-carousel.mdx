import React, { useEffect, useRef, useState } from "react";

const Carousel = ({ data }: any) => {
  const ITEM_NUMBER = data.length;

  const [currentIndex, setCurrentIndex] = useState(0);

  // Refs for the container and the slideItem
  const slideContainer = useRef<HTMLDivElement>(null);
  const slideItem = useRef<HTMLDivElement>(null);

  let width: any;
  let offset: any;
  let position: any;

  // This Function is only for mobileðŸ“±
  // Scroll function that calculate the offset
  const scroll = () => {
    offset = slideContainer?.current?.scrollLeft;

    if (offset && width) {
      position = Math.round(offset / width);
    }
    setCurrentIndex(position);
  };

  useEffect(() => {
    width = slideItem?.current?.clientWidth;
    if (window.innerWidth < 770) {
      slideContainer?.current?.addEventListener("scroll", scroll);
    }
  });

  // This is for the larger devices ðŸ–¥ï¸
  // Add the translate transition to slideContainer
  useEffect(() => {
    if (window.innerWidth > 770 && slideContainer.current) {
      slideContainer.current.style.width = `${700 * ITEM_NUMBER}px`;
      slideContainer.current.style.transform = `translateX(${
        -width * currentIndex - currentIndex * 30
      }px)`;
    }
  }, [ITEM_NUMBER, currentIndex, width]);

  const next = () => {
    setCurrentIndex(currentIndex + 1);
  };
  const prev = () => {
    setCurrentIndex(currentIndex - 1);
  };
  return (
    <div className="w-full max-w-[1200px]">
      <div className="w-full overflow-hidden">
        <div
          ref={slideContainer}
          className={`md:snap:none scrollbar-none mb-14 flex snap-x snap-mandatory space-x-8 overflow-scroll  transition duration-500   md:overflow-hidden md:pl-[29%]`}
        >
          {data.map((props: any, index: number) => {
            return (
                // add your code
            );
          })}
        </div>
      </div>
      <div className="flex items-center justify-center space-x-2">
        {Array.from({ length: ITEM_NUMBER }).map((_, index) => (
          <div
            key={`item-${index}`}
            className={
              currentIndex === index
                ? "carousel-index"
                : "carousel-index  opacity-20"
            }
          ></div>
        ))}
      </div>
      <button
        type="button"
        className="absolute top-0 left-0 z-30 flex items-center justify-center h-full px-4 cursor-pointer group focus:outline-none"
        onClick={prev}
        disabled={currentIndex === 0}
      >
        <span className="inline-flex items-center justify-center w-10 h-10 rounded-full bg-white/30 dark:bg-gray-800/30 group-hover:bg-white/50 dark:group-hover:bg-gray-800/60 group-focus:ring-4 group-focus:ring-white dark:group-focus:ring-gray-800/70 group-focus:outline-none">
          <svg
            className="w-4 h-4 text-white dark:text-gray-800"
            xmlns="http://www.w3.org/2000/svg"
            fill="none"
            viewBox="0 0 6 10"
          >
            <path
              stroke="currentColor"
              stroke-linecap="round"
              stroke-linejoin="round"
              stroke-width="2"
              d="M5 1 1 5l4 4"
            />
          </svg>
          <span className="sr-only">Previous</span>
        </span>
      </button>
      <button
        type="button"
        className="absolute top-0 right-0 z-30 flex items-center justify-center h-full px-4 cursor-pointer group focus:outline-none"
        onClick={next}
        disabled={currentIndex === ITEM_NUMBER - 1}
      >
        <span className="inline-flex items-center justify-center w-10 h-10 rounded-full bg-white/30 dark:bg-gray-800/30 group-hover:bg-white/50 dark:group-hover:bg-gray-800/60 group-focus:ring-4 group-focus:ring-white dark:group-focus:ring-gray-800/70 group-focus:outline-none">
          <svg
            className="w-4 h-4 text-white dark:text-gray-800"
            xmlns="http://www.w3.org/2000/svg"
            fill="none"
            viewBox="0 0 6 10"
          >
            <path
              stroke="currentColor"
              stroke-linecap="round"
              stroke-linejoin="round"
              stroke-width="2"
              d="m1 9 4-4-4-4"
            />
          </svg>
          <span className="sr-only">Next</span>
        </span>
      </button>
    </div>
  );
};

export default Carousel;
