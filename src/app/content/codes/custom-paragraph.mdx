---
 title: "Custom Paragraph"
---
 
```tsx

import React from 'react';

export type ParagraphProps = {
  paragraph: string;
  isPrimary?: boolean;
  className?: string;
};

function Paragraph({
  paragraph,
  isPrimary = false,
  className,
}: ParagraphProps) {
  return (
    <p
      className={
        isPrimary
          ? `mb-3 text-base leading-8 text-black lg:mb-5 lg:text-[20px]  ${className}`
          : `mb-3 text-sm leading-8 text-grey-2 lg:mb-5  lg:text-base ${className}`
      }
    >
      {paragraph}
    </p>
  );
}

export default Paragraph;


```
